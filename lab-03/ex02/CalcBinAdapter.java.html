package ex02;

public class CalcBinAdapter implements Calculadora<String> {

    private CalcBinaria calc;

    public CalcBinAdapter(CalcBinaria calc) {
        this.calc = calc;
    }

    @Override
    public String somar(String a, String b) {
        return  calc.somar(a, b);
    }

    @Override
    public String subtrair(String a, String b) {
        return calc.subtrair(a, b);
    }

    @Override
    public String multiplicar(String a, String b) {
        // Somente exemplo
        return calc.somar(a, b) + calc.subtrair(b,a);
    }

}
