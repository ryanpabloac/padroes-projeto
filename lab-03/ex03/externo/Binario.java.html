package ex03.externo;

import java.util.ArrayList;
import java.util.List;

public class Binario {

    private List<Byte> textoBin;

    public Binario() {
        textoBin = new ArrayList<>();
    }

    public void gravarCaractere(char caractere) {
        String sChar = String.valueOf(caractere);
        textoBin.add(Byte.parseByte(sChar));
    }

    public List<Byte> mostrarNinario() {
        return textoBin;
    }
}
